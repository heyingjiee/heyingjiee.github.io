// 节流。第一次触发后，延迟时间内再次触发无效
function throttle(fn,delay){
    let timer=null
    return function(..args){
      if(timer){
        return
    	}

      timer=setTimeoute(()=>{
          fn(...args)
        	clearTimeout(timer)
        	timer=null
      },dalay)
    }
}

// 防抖。第一次触发后，延迟时间内再次触发重新计时
function debounce(fn,delay){
   let timer=null
   return function(...args){
      if(timer){
        clearTimeout(timer)
      }
      timer=setTimeout(()=>{
         fn(...args)
         clearTimeout(timer)
         timer=null
       },delay)
   }
}

// -----------  递归实现  -----------  
function deepClone(target){
    if(typeof target!=='object'||target===null){
        return target;
    }

    const copyTarget=Array.isArray(target)?[]:{}

    for (let key in target){
        copyTarget[key]=deepClone(target[key]) // 递归位置，如果是基础类型就直接return值，如果是object就会递归
    }

    return copyTarget
}


// 循环引用会导致堆栈溢出，这里是改进版
// 核心是每次递归的是对象类型，就使用WeakMap保存
function deepClonePlus(target,hash){
    if(target!=='object'||target===null){
        return target
    }

    if (hash.has(target)){
        return hash.get(target)
    }

    const copyTarget= Array.isArray(target)?[]:{}

    hash.set(target,copyTarget) // copyTarget就是复制本层对象

    for (let key in target){
        copyTarget[key]=deepClone(target[key],hash)
    }

    return target
}

promise.all

js
function pAll (_promises) {
  return new Promise((resolve, reject) => {
    // Iterable => Array
    const promises = Array.from(_promises)
    // 结果用一个数组维护
    const r = []
    const len = promises.length
    let count = 0
    for (let i = 0; i < len; i++) {
      // Promise.resolve 确保把所有数据都转化为 Promise
      // 这里有个语法 Promise.resolve
      //    参数是promise，会直接返回这个promise对象
      //    参数是非promise，会封装为promise对象
      Promise.resolve(promises[i]).then(o => { 
        // 因为 promise 是异步的，保持数组一一对应
        r[i] = o;
 
        // 如果数组中所有 promise 都完成，则返回结果数组
        if (++count === len) {
          resolve(r)
        }
        // 当发生异常时，直接 reject
      }).catch(e => reject(e))
    }
  })
}

function request(){
  const res=await axios.get('/xxx',{ responseType: 'blob'});
  // 正则匹配名字。服务端一般会设置：disposition: `attachment;filename="bg.jpg"
  const filename= res.headers['content-disposition'].match(/filename="(.*)"$/)[1] 
  console.log(res,filename);
  download(res.data,'application/zip',filename)
}

function download(data,type,filename){
  const blob=  new Blob([data],{type})
  const url =  URL.createObjectURL(blob)

  const downloadEle= document.createElement('a')
  downloadEle.href=url
  downloadEle.style.display='none'
  downloadEle.download=filename
  document.body.appendChild(downloadEle)
  downloadEle.click()
  document.body.removeChild(downloadEle)
  URL.revokeObjectURL(url)

}

https握手细节

js
首先由客户端向服务器端发送使用的协议的版本号、一个随机数和可以使用的加密方法。服务器端收到后，确认加密的方法，也向客户端发送一个随机数和自己的数字证书。客户端收到后，首先检查数字证书是否有效，如果有效，则再生成一个随机数，并使用证书中的公钥对随机数加密，然后发送给服务器端，并且还会提供一个前面所有内容的 hash 值供服务器端检验。服务器端接收后，使用自己的私钥对数据解密，同时向客户端发送一个前面所有内容的 hash 值供客户端检验。这个时候双方都有了三个随机数，按照之前所约定的加密方法，使用这三个随机数生成一把秘钥，以后双方通信前，就使用这个秘钥对数据进行加密后再传输。

性能指标 ​

  • 首屏加载时间First Contentful Paint（FCP）：首次内容绘制时间，指浏览器首次绘制页面中至少一个文本、图像、非白色背景色的canvas/svg元素等的时间，代表页面首屏加载的时间点。
  • 首次绘制时间First Paint（FP）：首次绘制时间，指浏览器首次在屏幕上渲染像素的时间，代表页面开始渲染的时间点。
  • 最大内容绘制时间Largest Contentful Paint（LCP）：最大内容绘制时间，指页面上最大的可见元素（文本、图像、视频等）绘制完成的时间，代表用户视觉上感知到页面加载完成的时间点。
  • 用户可交互时间Time to Interactive（TTI）：可交互时间，指页面加载完成并且用户能够与页面进行交互的时间，代表用户可以开始操作页面的时间点。
  • 页面总阻塞时间Total Blocking Time (TBT)：页面上出现阻塞的时间，指在页面变得完全交互之前，用户与页面上的元素交互时出现阻塞的时间。TBT应该尽可能小，通常应该在300毫秒以内。
  • 搜索引擎优化Search Engine Optimization (SEO)：网站在搜索引擎中的排名和可见性。评分范围从0到100，100分表示网站符合所有SEO最佳实践。

vite打包：将项目asset目录下的资源文件上传到CDN，通过renderBuiltUrl将index.html中引入资源文件的地址改为CDN地址

ts
import { fileURLToPath, URL } from 'node:url'
import { defineConfig, loadEnv } from 'vite'
import vue from '@vitejs/plugin-vue'

// https://vitejs.dev/config/
export default defineConfig(({ mode }) => {
  // eslint-disable-next-line no-undef
  const env = loadEnv(mode, process.cwd())

  const config = {
    plugins: [vue()],
    resolve: {
      alias: {
        '@': fileURLToPath(new URL('./src', import.meta.url))
      }
    }
  }

  if (env.VITE_APP_TARGET === 'production') {
    config.experimental = {
      // 静态资源放到 cdn
      renderBuiltUrl(filename, { type }) {
        console.log('renderBuiltUrl filename:',filename,type)
        if (type === 'asset') {
          return `//static1.bitautoimg.com/yc-miniprogram/ailive/${filename}`
        }
      }
    }
  }

  return config
})
